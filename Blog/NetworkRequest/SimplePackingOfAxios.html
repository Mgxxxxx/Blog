# 对Axios的简单封装

# 创建一个Axios实例，并简单配置

import axios form "axios"

const request = axios.create({
    baseURL: 'http://47.119.115.208:15656',
    headers: {
        "Content-Type": "application/json; charset=utf-8; */*",
    },
    timeout: 2000,
})

# 使用创建好的实例 "request" 进行get和post请求

有两种方式可以进行请求，一种是 "request(config)" ，"config"是一个对象，里面可以配置"method"，"url"等配置信息。另一种是使用请求发放别名来进行请求，例如 "axios.get(url[, config])" 和 "axios.post(url[, data[, config]])" 。一下是请求图片和上传图片的例子：

function getImg(img_id) {
    return request.get(
        "/getimg", {
            params: {
                img_id
            },
            responseType: "blob",	//设置响应的类型为blob，方便转化为图片
        }
    )
}

function uploadImg(data) {
    return request.post("/uploadimg", data, {
        headers: {
            "Content-Type": "multipart/form-data"	//
        },
    })
}
